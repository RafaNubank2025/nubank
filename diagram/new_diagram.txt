Table country {
  country_id uuid [pk]
  country varchar(128) 
  created_at timestamp
  updated_at timestamp
}

Table state {
  state_id uuid [pk]
  state varchar(128)
  country_id uuid
  created_at timestamp
  updated_at timestamp
}

Table city {
  city_id int [pk]
  city varchar(256)
  state_id uuid
  created_at timestamp
  updated_at timestamp
}

Table customers {
  customer_id uuid [pk]
  first_name varchar(128)
  last_name varchar(128)
  city_id int
  identification_number int
  created_at timestamp
  updated_at timestamp
}

Table accounts {
  account_id uuid [pk]
  customer_id uuid
  status varchar(128)
  account_branch varchar(128)
  account_check_digit varchar(128)
  account_number varchar(128)
  created_at timestamp
  updated_at timestamp
}

Table transactions {
  transaction_id uuid [pk]
  account_id uuid
  transaction_type varchar(128) // ('PIX', 'TED', 'DOC', 'BOLETO', 'CARTAO', 'EMPRESTIMO')
  transaction_direction varchar(128) // ('IN', 'OUT')
  amount float
  transaction_status varchar(128) // ('COMPLETED', 'FAILED')
  transaction_time_id int
  created_at timestamp
  updated_at timestamp
}

Table financial_products {
  product_id uuid [pk]
  account_id uuid
  product_type varchar(128) // ('CARTAO_CREDITO', 'SEGURO_VIDA', 'EMPRESTIMO')
  product_status varchar(128) // ('ACTIVE', 'CANCELLED')
  created_at timestamp
  updated_at timestamp
}

// Tabelas de tempo
Table d_month {
  month_id int [pk]
  action_month int
}

Table d_year {
  year_id int [pk]
  action_year int
}

Table d_time {
  time_id int [pk]
  action_timestamp timestamp
  week_id int
  month_id int
  year_id int
  weekday_id int
}

Table d_week {
  week_id int [pk]
  action_week int
}

Table d_weekday {
  weekday_id int [pk] 
  action_weekday varchar(128)
}

Ref: accounts.customer_id > customers.customer_id
Ref: transactions.account_id > accounts.account_id
Ref: financial_products.account_id > accounts.account_id
